    \section{Actions mécaniques}

        \subsection*{Moment d'une force}
        \[
        \vec M_O(F) = \vec{OA} \wedge \vec F
        \]
        Avec A le point d'application de la force.

        \subsection*{Transport du moment (BABAR)}
        \[
        \vec M(B) = \vec M(A) + \vec{BA} \wedge \vec R
        \]
        
        \subsection*{Propriété}
        \[
        \vec M \perp \vec R \quad \text{(glisseur, automoment nul)}
        \]

    % ======================================================
    \section{Torseur d'action mécanique}

        \[
        \{T\}_A =
        \begin{Bmatrix}
        \vec R \\
        \vec M(A)
        \end{Bmatrix}
        \]

        \subsection*{Invariants}
        \begin{itemize}
            \item Invariant vectoriel : $\vec R$
            \item Automoment : $\vec R \cdot \vec M(A) = \text{cte}$
        \end{itemize}

        \subsection*{Torseurs particuliers}
            $$ \text{Glisseur}
            \begin{Bmatrix}
                \vec R \\
                \vec 0
            \end{Bmatrix}
            \qquad
            \text{Couple}
            \begin{Bmatrix}
                \vec 0 \\
                \vec C
            \end{Bmatrix}
            $$

    \section{Cinématique du solide}

        \subsection*{Transport des vitesses (BABAR)}
        Le champ des vitesses est un champ de moments :
        \[
        \vec V_B = \vec V_A + \vec{BA} \wedge \vec\Omega
        \]

        Pour un solide $S$ en mouvement par rapport à un référentiel $R$ :
        \[
        \vec V(B, S/R) = \vec V(A, S/R) + \vec{BA} \wedge \vec\Omega_{S/R}
        \]


        \subsection*{Transport des accélérations}
        \[
        \vec A_P =
        \vec A_O
        + \dv{\vec\Omega}{t} \wedge \vec{OP}
        + \vec\Omega \wedge (\vec\Omega \wedge \vec{OP})
        \]

    \section{Composition des vitesses}

        =

\section{Conditions géométriques des liaisons}
 \begin{multicols}{2}

    \subsection{Liaison ponctuelle (normale $\vec x_i$)}
        Créer une liaison ponctuelle revient à contraindre le point $O_j$ à
        appartenir à la surface du solide $S_i$.

        Condition géométrique :
        \[
        \vec{M_iO_j} \cdot \vec x_i = 0
        \]


    \subsection{Liaison sphérique (rotule, centre $O$)}
        Les points $O_i$ et $O_j$ sont confondus :
        \[
        \vec{O_iO_j} = \vec 0
        \]


    \subsection{Liaison plane (appui plan, normale $\vec x_i$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} \cdot \vec x_i = 0
        \]
        Et les normales des deux solides sont colinéaires (1 seule liberté de rotation) :
        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]


    \subsection{Liaison pivot-glissant / glissière (axe $(O,\vec x_i)$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} = \lambda \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec{O_iO_j} \cdot \vec y_i = 0 \\
        \vec{O_iO_j} \cdot \vec z_i = 0
        \end{cases}
        \]

        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]

        Si c'est une glissière alors absence de rotation autour de $\vec x_i$ :
        \[
        \vec y_j \cdot \vec y_i = 1
        \]

    \subsection{Liaison pivot (axe $(O,\vec x_i)$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} = \vec 0
        \]

        (3 équations scalaires)

        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]

    \end{multicols}

    \section{Géométrie des masses}
        \begin{multicols}{2}
            
        % ------------------------------------------------------
        \subsection{Masse d'un solide}

            Un solide $S$ est constitué d'éléments de masse $dm$ :

            \[
            m = \int_S dm
            \]

            Si le solide est homogène (masse volumique $\rho$ constante) :

            \[
            m = \int_S \rho \, dV
            \]

            En coordonnées cartésiennes :

            \[
            m = \rho \iiint_S dx\,dy\,dz
            \]


        % ------------------------------------------------------
        \subsection{Centre de masse (centre d'inertie)}

            Le centre de masse $G$ est le barycentre des points $P$ pondérés par
            leur masse :

            \[
            \, \vec{OG} = \frac{1}{m} \int_{P\in S} \vec{OP} \, dm
            \]

            Coordonnées du centre de masse :

            \[
            x_G = \frac{1}{m} \int_S x\,dm
            \quad
            y_G = \frac{1}{m} \int_S y\,dm
            \quad
            z_G = \frac{1}{m} \int_S z\,dm
            \]



        % ------------------------------------------------------
        \subsection{Propriétés de symétrie}

            \begin{itemize}
                \item Si le solide possède un plan de symétrie, $G$ appartient à ce plan.
                \item Si le solide est de révolution, $G$ appartient à l'axe.
            \end{itemize}


        % ------------------------------------------------------
        \subsection{Centre de masse d'un ensemble de solides}

            Pour un système de solides $S_i$ :

            \[
            \vec{OG} =
            \frac{\sum_i m_i \vec{OG_i}}
            {\sum_i m_i}
            \]

            avec $G_i$ le centre de masse de chaque solide.


        % ------------------------------------------------------
        \subsection{Centre de masse et centre de gravité}

            Le torseur du poids est :

            \[
            \left\{T_{pes \rightarrow S}\right\}_G
            =
            \begin{Bmatrix}
            m\vec g \\
            \vec 0
            \end{Bmatrix}_G
            \]

            Si le champ de pesanteur $\vec g$ est uniforme :

            \[
            \text{Centre de gravité} = \text{Centre de masse}
            \]

        % \end{multicols}

            % ======================================================
        \section{Torseur cinématique}

            Le mouvement d'un solide $S$ par rapport à $R$ est décrit par le torseur :

            \[
            \left\{V(S/R)\right\}_A =
            \begin{Bmatrix}
            \vec \Omega(S/R) \\
            \vec V(A,S/R)
            \end{Bmatrix}
            \]

            \begin{itemize}
                \item $\vec \Omega$ : vecteur rotation
                \item $\vec V(A)$ : vitesse d'un point du solide
            \end{itemize}

            \subsection*{Transport du torseur cinématique}

            \[
            \vec V(B) =
            \vec V(A)
            + \vec \Omega \wedge \vec{AB}
            \]

    \section{Torseur cinétique}

            \[
            \left\{C(S/0)\right\}_Q =
            \begin{Bmatrix}
            \vec p (S/0) \\
            \vec \sigma (Q, S/0)
            \end{Bmatrix}_Q
            =
            \begin{Bmatrix}
            \int_{P\in S} \vec V(P, S/0) \, dm(P) \\
            \int_{P\in S} \vec{QP} \wedge \vec{V}(P, S/0) \, dm(P)
            \end{Bmatrix}_Q
            \]

            \begin{itemize}
                \item $\vec p$ : résultante cinétique \quad $\bm{\vec p (S/0) = m \vec V (G/0)}$
                \item $\vec \sigma$ : moment cinétique. En un point fixe dans $R_0$ ou en G: \\
                $\bm{\vec\sigma (G, S/0) =} \Inertie{G}{S} \bm{\, \vec\Omega_{S/0}}$
            \end{itemize}

            \subsection*{Formule de Transport}

                $$ \vec \sigma (B, S/0) = \vec \sigma (A, S/0) + \vec{BA} \wedge \vec p(S/0) $$

        \section{Torseur Dynamique}

            $$
                \left\{D_{S/0}\right\}_Q =
                \begin{Bmatrix}
                    \vec D (S/0) \\
                    \vec \delta (Q, S/0)
                \end{Bmatrix}
                =
                \begin{Bmatrix}
                    \int_{P\in S} \vec A(P, S/0) \, dm(P) \\
                    \int_{P\in S} \vec{QP} \wedge \vec A(P, S/0) \, dm(P)
                \end{Bmatrix}_Q
            $$

            \begin{itemize}
                \item $\vec D$ : résultante dynamique
                $$ \bm{\vec D(S/0) = m \,\vec A(G, S/0) \qquad \vec D(S/0) = \frac{d}{dt}\vec p(S/0)} $$
                \item $\vec \delta$ : moment dynamique  $ \bm{\vec \delta (G, S/0) = \frac{d}{dt} \vec \sigma (G, S/0)} $ \\
                En un point $Q$ quelconque :
                $$ \bm{\vec \delta (Q, S/0) = \frac{d}{dt} \vec \sigma (Q, S/0)
                  + m \vec V(Q, S/0) \wedge \vec V(G/0)} $$
            \end{itemize}

            \subsection*{Formule de Transport}                
                  $$ \vec \delta (B, S/0) = \vec \delta (A, S/0) + \vec{BA} \wedge \vec D(S/0) $$
        \end{multicols}

        \begin{multicols}{2}
        [
        \section{Inertie du solide}
        ]
            L'inertie décrit la répartition de masse du solide
            par rapport à un point.

            \subsection{Moment d'inertie}

            Par rapport à un axe :

            \[
            I_\Delta = \int_S r^2\, dm
            \]

            avec $r$ la distance à l'axe.


        % ------------------------------------------------------
        \subsection{Tenseur d'inertie}

            Le tenseur d'inertie du solide $S$ au point $O$ est :

            \[
            \Inertie{O}{S} =
            \begin{bmatrix}
            A  & -F & -E \\
            -F &  B & -D \\
            -E & -D &  C
            \end{bmatrix}
            \quad \text{(notation de Binet)}
            \]

            avec les moments d'inertie :
            \[
            A = \int (y^2+z^2)\,dm, \quad
            B = \int (x^2+z^2)\,dm, \quad
            C = \dots
            \]

            et les produits d'inertie :
            \[
            D = \int yz\,dm, \quad
            E = \int xz\,dm, \quad
            F = \int xy\,dm
            \]

        \noindent $A$, $B$, $C$ sont les moments d'inertie et \\
        $D$, $E$, $F$ les produits d'inertie.\newline

        Lorsque le solide $S$ admet des plans de symétrie, certains produits d'inertie s'annulent (s'ils sont exprimés en un point appartenant au plan de symétrie) :

        \begin{itemize}
            \item $(O, x, y)$ est PS $\rightarrow$ $E$ et $D$ s'annulent.
            \item $(O, x, z)$ est PS $\rightarrow$ $F$ et $D$ s'annulent.
            \item $(O, y, z)$ est PS $\rightarrow$ $F$ et $E$ s'annulent.
        \end{itemize}

        \medskip

        Lorque le solide est un solide de révolution, les produits d'inerties exprimés en un point appartenant à l'axe de révolution s'annulent.


        % ------------------------------------------------------
        \subsection{Théorème de Huygens}

            Si on pose 
            $$ \vec{QG} = \begin{pmatrix}
                a \\ b \\ c \\
                \end{pmatrix} $$
            alors 
            $$ \Inertie{Q}{S} = \Inertie{G}{S} + m \begin{pmatrix}
                b^2+c^2 & -ab & -ac \\
                -ab & a^2+c^2 & -bc \\
                -ac & -bc & a^2+b^2
            \end{pmatrix} $$

            \sum \vec F_{\text{ext}} = m \vec A_G = \vec D
            \]

            TMD:
            $$ \sum \vec{\mathcal{M}} = \vec \delta $$

       \section{Lois de comportement des amortisseurs visqueux}

            Un amortisseur visqueux développe une action mécanique \textbf{proportionnelle à la vitesse} et \textbf{dissipe de l'énergie}.

            \textbf{Amortisseur de traction/compression}:
            \[
            \{T_{A/2}\} = \begin{Bmatrix} -b\,\dot{\lambda}\,\vec{u} \\ \vec{0} \end{Bmatrix}_A
            \]
            avec $\lambda = \overrightarrow{A_1A_2}\cdot\vec{u}$ (longueur courante), et $b$ le coefficient d'amortissement visqueux en $ N s / m $.

            \textbf{Amortisseur de torsion}:
            \[
            \vec{M}_{A_2} = -b\,\dot{\theta}\,\vec{u}\, , \quad \text{avec } \dot{\theta} = \vec{\Omega}_{S2/S1}\cdot\vec{u}
            \]
