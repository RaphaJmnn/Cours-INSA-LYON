    \section{Actions mécaniques}

        \subsection*{Moment d'une force}
        \[
        \vec M_O(F) = \vec{OA} \wedge \vec F
        \]
        Avec A le point d'application de la force.

        \subsection*{Transport du moment (BABAR)}
        \[
        \vec M(B) = \vec M(A) + \vec{BA} \wedge \vec R
        \]
        
        \subsection*{Propriété}
        \[
        \vec M \perp \vec R \quad \text{(glisseur, automoment nul)}
        \]

    % ======================================================
    \section{Torseur d'action mécanique}

        \[
        \{T\}_A =
        \begin{Bmatrix}
        \vec R \\
        \vec M(A)
        \end{Bmatrix}
        \]

        \subsection*{Invariants}
        \begin{itemize}
            \item Invariant vectoriel : $\vec R$
            \item Automoment : $\vec R \cdot \vec M(A) = \text{cte}$
        \end{itemize}

        \subsection*{Torseurs particuliers}
        \begin{itemize}
            \item Glisseur :
            \[
            \begin{Bmatrix}
            \vec R \\
            \vec 0
            \end{Bmatrix}
            \]
            \item Couple :
            \[
            \begin{Bmatrix}
            \vec 0 \\
            \vec C
            \end{Bmatrix}
            \]
        \end{itemize}

    \section{Axe central}

        \subsection*{Définition}
        \[
        \vec M(I) = \lambda \vec R
        \]

        \subsection*{Formule}
        \[
        \vec{AI} =
        \frac{\vec R \wedge \vec M(A)}{\|\vec R\|^2}
        + k \vec R
        \]

        \subsection*{Propriétés}
        \begin{itemize}
            \item Moment constant sur l'axe
            \item Moment minimum sur l'axe
        \end{itemize}

    \section{Statique}

        \subsection*{Principe Fondamental de la Statique}
        \[
        \sum \vec F_{ext \rightarrow S} = \vec 0
        \qquad
        \sum \vec M_{ext \rightarrow S}(A) = \vec 0
        \]

    \section{Cinématique du solide}

        \subsection*{Transport des vitesses (BABAR)}
        Le champ des vitesses est un champ de moments :
        \[
        \vec V_P = \vec V_O + \vec\Omega \wedge \vec{OP}
        \]

        Pour un solide $S$ en mouvement par rapport à un référentiel $R$ :
        \[
        \vec V(B, S/R) = \vec V(A, S/R) + \vec{BA} \wedge \vec\Omega(S/R)
        \]


        \subsection*{Transport des accélérations}
        \[
        \vec A_P =
        \vec A_O
        + \dv{\vec\Omega}{t} \wedge \vec{OP}
        + \vec\Omega \wedge (\vec\Omega \wedge \vec{OP})
        \]

    % ======================================================
    \section{Composition des vitesses}

        \[
        \vec V(P, S/R_0)
        =
        \vec V(P, S/R_1)
        +
        \vec V(P, R_1/R_0)
        \]

    % ======================================================
    \section{Composition des accélérations}

        \begin{align*}  
            \vec A(P, S/R_0) = \vec A(P, S/R_1) + \vec A(P, R_1/R_0) \\
                                + 2 \vec\Omega(R_1/R_0) \wedge \vec V(P, S/R_1)
        \end{align*}

        \textit{(Terme de Coriolis)}

    % ======================================================
    \section{Dérivée d'un vecteur tournant}

        \[
        \left( \frac{d\vec u}{dt} \right)_{R_0}
        =
        \left( \frac{d\vec u}{dt} \right)_{R_1}
        +
        \vec\Omega(R_1/R_0) \wedge \vec u
        \]

    \section{Roulement sans glissement}

        \subsection{Hypothèse de roulement sans glissement}
        Au point de contact $I$ entre deux solides $S_1$ et $S_2$ :
        \[
        \vec V(I, S_1/S_2) = \vec 0
        \]

        Cette hypothèse impose :
        \[
        \vec V(I, S_1/R_0) = \vec V(I, S_2/R_0)
        \]

\section{Conditions géométriques des liaisons}

    \subsection{Liaison ponctuelle (normale $\vec x_i$)}
        Créer une liaison ponctuelle revient à contraindre le point $O_j$ à
        appartenir à la surface du solide $S_i$.

        Condition géométrique :
        \[
        \vec{M_iO_j} \cdot \vec x_i = 0
        \]


    \subsection{Liaison sphérique (rotule, centre $O$)}
        Les points $O_i$ et $O_j$ sont confondus :
        \[
        \vec{O_iO_j} = \vec 0
        \]


    \subsection{Liaison plane (appui plan, normale $\vec x_i$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} \cdot \vec x_i = 0
        \]
        Et les normales des deux solides sont colinéaires (1 seule liberté de rotation) :
        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]


    \subsection{Liaison pivot-glissant / glissière (axe $(O,\vec x_i)$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} = \lambda \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec{O_iO_j} \cdot \vec y_i = 0 \\
        \vec{O_iO_j} \cdot \vec z_i = 0
        \end{cases}
        \]

        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]

        Si c'est une glissière alors absence de rotation autour de $\vec x_i$ :
        \[
        \vec y_j \cdot \vec y_i = 1
        \]

    \subsection{Liaison pivot (axe $(O,\vec x_i)$)}

        Conditions géométriques :
        \[
        \vec{O_iO_j} = \vec 0
        \]

        (3 équations scalaires)

        \[
        \vec x_j = \vec x_i
        \quad \Rightarrow \quad
        \begin{cases}
        \vec x_j \cdot \vec y_i = 0 \\
        \vec x_j \cdot \vec z_i = 0
        \end{cases}
        \]

    \section{Géométrie des masses}

        La géométrie des masses décrit :
        \begin{itemize}
            \item la masse du solide
            \item la position du centre de masse
            \item la répartition de masse (inertie)
        \end{itemize}

        % ------------------------------------------------------
        \subsection{Masse d'un solide}

            Un solide $S$ est constitué d'éléments de masse $dm$ :

            \[
            m = \int_S dm
            \]

            Si le solide est homogène (masse volumique $\rho$ constante) :

            \[
            m = \int_S \rho \, dV
            \]

            En coordonnées cartésiennes :

            \[
            m = \rho \iiint_S dx\,dy\,dz
            \]


        % ------------------------------------------------------
        \subsection{Centre de masse (centre d'inertie)}

            Le centre de masse $G$ est le barycentre des points $P$ pondérés par
            leur masse :

            \[
            m \, \vec{OG} = \int_S \vec{OP} \, dm
            \]

            Coordonnées du centre de masse :

            \[
            x_G = \frac{1}{m} \int_S x\,dm
            \qquad
            y_G = \frac{1}{m} \int_S y\,dm
            \qquad
            z_G = \frac{1}{m} \int_S z\,dm
            \]

            Si le solide est homogène :

            \[
            x_G = \frac{1}{V}\int_S x\,dV
            \]


        % ------------------------------------------------------
        \subsection{Propriétés de symétrie}

            \begin{itemize}
                \item Si le solide possède un plan de symétrie, $G$ appartient à ce plan.
                \item Si le solide est de révolution, $G$ appartient à l'axe.
                \item Un solide avec trois plans de symétrie a son centre au milieu.
            \end{itemize}


        % ------------------------------------------------------
        \subsection{Centre de masse d'un ensemble de solides}

            Pour un système de solides $S_i$ :

            \[
            \vec{OG} =
            \frac{\sum_i m_i \vec{OG_i}}
            {\sum_i m_i}
            \]

            avec $G_i$ le centre de masse de chaque solide.


        % ------------------------------------------------------
        \subsection{Centre de masse et centre de gravité}

            Le torseur du poids est :

            \[
            \left\{T_{pes \rightarrow S}\right\}
            =
            \begin{Bmatrix}
            m\vec g \\
            \vec 0
            \end{Bmatrix}_G
            \]

            Si le champ de pesanteur $\vec g$ est uniforme :

            \[
            \text{Centre de gravité} = \text{Centre de masse}
            \]

            % ======================================================
        \section{Torseur cinématique}

            Le mouvement d'un solide $S$ par rapport à $R$ est décrit par le torseur :

            \[
            \left\{V(S/R)\right\}_A =
            \begin{Bmatrix}
            \vec \Omega(S/R) \\
            \vec V(A,S/R)
            \end{Bmatrix}
            \]

            \begin{itemize}
                \item $\vec \Omega$ : vecteur rotation
                \item $\vec V(A)$ : vitesse d'un point du solide
            \end{itemize}

            \subsection*{Transport du torseur cinématique}

            \[
            \vec V(B) =
            \vec V(A)
            + \vec \Omega \wedge \vec{AB}
            \]

    \section{Torseur cinétique}

            \[
            \left\{C(S/0)\right\} =
            \begin{Bmatrix}
            \vec p (S/0) \\
            \vec \sigma (Q, S/0)
            \end{Bmatrix}_Q
            =
            \begin{Bmatrix}
            \int_{P\in S} \vec V(P, S/0) \, dm(P) \\
            \int_{P\in S} \vec{QP} \wedge \vec{V}(P, S/0) \, dm(P)
            \end{Bmatrix}_Q
            \]

            \begin{itemize}
                \item $\vec p$ : résultante cinétique
                \item $\vec \sigma$ : moment cinétique (si $Q \equiv G \quad \Rightarrow \quad \vec \sigma (G, \, S/0)= \overset{=}{I}(G, S) \vec \Omega_{S/0}$)
            \end{itemize}

        \section{Torseur Dynamique}

            \[
            \left\{D_{S/0}\right\} =
            \begin{Bmatrix}
            \vec D (S/0) \\
            \vec \delta (Q, S/0)
            \end{Bmatrix}
            =
            \begin{Bmatrix}
            \int_{P\in S} \vec A(P, S/0) \, dm(P) \\
            \int_{P\in S} \vec{QP} \wedge \vec A(P, S/0) \, dm(P)
            \end{Bmatrix}_Q
            \]

            \begin{itemize}
                \item $\vec D$ : résultante dynamique ($\vec D(S/0) = m(S) \, \vec A(G, \, S/0)$ et $\vec D(S/0) = \frac{d}{dt}\vec p(S/0))$)
                \item $\vec \delta$ : moment dynamique (si $Q \equiv G \quad \Rightarrow \quad \vec \delta (G, \, S/0) = \frac{d}{dt} \vec \sigma (G, \, S/0)$)
    
                et en un point $Q$ quelconque: $\vec \delta (Q, \, S/0) = \frac{d}{dt} \vec \sigma (Q, \, S/0) + \vec V(Q, \, S/0) \wedge \vec p(S/0)$
            \end{itemize}

        % ======================================================
        \section{Inertie du solide}

            L'inertie décrit la répartition de masse du solide
            par rapport à un point.

            \subsection{Moment d'inertie}

            Par rapport à un axe :

            \[
            I_\Delta = \int_S r^2 dm
            \]

            avec $r$ la distance à l'axe.


        % ------------------------------------------------------
        \subsection{Tenseur d'inertie}

            Le tenseur d'inertie du solide $S$ au point $O$ est :

            \[
            \mathbf{\overset{=}{I}(O, S)} =
            \begin{bmatrix}
            I_{xx} & -I_{xy} & -I_{xz} \\
            -I_{xy} & I_{yy} & -I_{yz} \\
            -I_{xz} & -I_{yz} & I_{zz}
            \end{bmatrix}
            = \begin{bmatrix}
                A & -F & -E \\
                -F & B & -D \\
                -E & -D & C
             \end{bmatrix}
            \quad \text{Notation de Binet}\]

            avec :

            $$
            I_{xx} = A = \int (y^2+z^2)\,dm, \quad I_{yy} = B = \int (x^2+z^2)\,dm, \quad I_{zz} = C = \int (x^2+y^2)\,dm
            $$

            $$
            I_{xy} = -F = \int xy\,dm, \quad I_{yz} = -D = \int yz\,dm, \quad I_{xz} = -E = \int xz\,dm
            $$

        $A$, $B$ et $C$ sont les moments d'inertie et $D$, $E$ et $F$ les produits d'inertie.

        Lorsque le solide $S$ admet des plans de symétrie, alors certains produits de symétrie s'annulent :

        \begin{itemize}
            \item (O, x, y) est PS $\rightarrow$ $E$ et $D$ s'annulent.
            \item (O, x, z) est PS $\rightarrow$ $F$ et $D$ s'annulent.
            \item (O, y, z) est PS $\rightarrow$ $F$ et $E$ s'annulent.
        \end{itemize}


        % ------------------------------------------------------
        \subsection{Théorème de Huygens}

            Translation du tenseur d'inertie :

            \[
            \mathbf I_O =
            \mathbf I_G +
            m(\|\vec{OG}\|^2 \mathbf I -
            \vec{OG}\vec{OG}^{\,T})
            \]

            Pour un axe parallèle :

            \[
            I_O = I_G + md^2
            \]


        % ------------------------------------------------------
        \subsection{Principe Fondamental de la Dynamique}

            Le torseur des actions extérieures est égal
            au torseur dynamique :

            \[
            \left\{T_{ext \rightarrow S}\right\}_A
            =
            \left\{D(S/R)\right\}_A
            \]

            soit :

            \[
            \sum \vec F = m \vec A_G
            \]

            \[
            \sum \vec M(A) =
            \vec M_d(A)
            \]


        % ------------------------------------------------------
        \subsection{Cas particuliers}

            \textbf{Translation}

            \[
            \vec M_d(G)=\vec 0
            \]

            \[
            \sum \vec F = m\vec A_G
            \]

            \textbf{Rotation autour d'un axe fixe}

            \[
            \sum M_\Delta =
            I_\Delta \dot{\omega}
            \]



        % ======================================================
        \section{Énergie cinétique}

            \subsection{Expression générale}

            \[
            E_c =
            \frac12 m V_G^2 +
            \frac12 \vec \Omega \cdot
            \mathbf I_G \vec \Omega
            \]

            \subsection{Translation}

            \[
            E_c = \frac12 m V^2
            \]

            \subsection{Rotation autour d'un axe}

            \[
            E_c =
            \frac12 I_\Delta \omega^2
            \]
